import java.util.Scanner;

public class TokenRing {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input number of processes
        System.out.print("Enter number of nodes: ");
        int n = sc.nextInt();

        // Input sender, receiver and data
        System.out.print("Enter sender: ");
        int sender = sc.nextInt();

        System.out.print("Enter receiver: ");
        int receiver = sc.nextInt();

        System.out.print("Enter data: ");
        int data = sc.nextInt();

        // Step 1: Pass token from 0 to sender
        System.out.println("\nToken passing:");
        int i = 0;
        while (i != sender) {
            System.out.print(i + " -> ");
            i = (i + 1) % n;
        }
        System.out.println(sender);

        // Step 2: Sender sends data
        System.out.println("Sender " + sender + " sends data: " + data);

        // Step 3: Data moves until receiver
        i = (sender + 1) % n;
        while (i != receiver) {
            System.out.println("Data forwarded by " + i);
            i = (i + 1) % n;
        }

        // Step 4: Receiver gets data
        System.out.println("Receiver " + receiver + " received data: " + data);

        sc.close();
    }
}
